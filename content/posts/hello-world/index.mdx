## Testing Code Highlighting

Here's a TypeScript example to test Shiki syntax highlighting:

```typescript
// A simple TypeScript function to demonstrate syntax highlighting
function greet(name: string): string {
  return `Hello, ${name}! 👋`;
}

interface BlogPost {
  title: string;
  date: Date;
  tags: string[];
  published: boolean;
}

// Arrow function with destructuring
const formatPost = ({ title, date }: BlogPost): string => {
  return `${title} - ${date.toLocaleDateString()}`;
```
